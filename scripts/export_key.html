E2EE 密钥导出

打开浏览器控制台 (F12)，粘贴以下代码运行，下载私钥文件：

// 导出当前用户的 E2EE 私钥
(async () => {
  const db = await new Promise((r, x) => {
    const req = indexedDB.open('chat-local-keys', 1);
    req.onsuccess = () => r(req.result);
    req.onerror = () => x(req.error);
  });
  const tx = db.transaction('keys', 'readonly');
  const key = await new Promise((r, x) => {
    const req = tx.objectStore('keys').get('my-keypair');
    req.onsuccess = () => r(req.result);
    req.onerror = () => x(req.error);
  });
  if (!key) { console.log('未找到私钥！请先在应用中打开一次聊天'); return; }

  const exported = await crypto.subtle.exportKey('pkcs8', key);
  const base64 = btoa(String.fromCharCode(...new Uint8Array(exported)));

  // 下载文件
  const blob = new Blob([base64], {type:'text/plain'});
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = 'e2ee_private_key.txt';
  a.click();
  console.log('✅ 私钥已下载！');
})();
